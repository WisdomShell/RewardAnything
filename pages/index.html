Quick Start

Get started with RewardAnything in just a few lines of code

Installation

pip install rewardanything

Basic Usage

import rewardanything

# Load model locally
reward_model = rewardanything.from_pretrained(
    "zhuohaoyu/RewardAnything-8B-v1",
    device="cuda",
    torch_dtype="auto"
)

# Define your evaluation principle
principle = "I prefer clear, concise and helpful responses."

# Your evaluation data
responses = {
    "model_a": "Python is a programming language...",
    "model_b": "Python is a snake."
}

# Get comprehensive evaluation
result = reward_model.judge(
    principle=principle,
    prompt="What is Python?",
    responses=responses
)

print(f"Best to worst: {result.ranking}")
